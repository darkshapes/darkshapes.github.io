 28    @staticmethod
 29    def show_diffusers_tasks(code_name: str, class_name: str | None = None) -> list[str]:
 30        """Return Diffusers task pipes based on package-specific query\n
 31        :param class_name: To find task pipes from a Diffusers class pipe, defaults to None
 32        :param code_name: To find task pipes from a Transformers class pipe, defaults to None
 33        :return: A list of alternate class pipelines derived from the specified class"""
 34
 35        if class_name:
 36            from diffusers.pipelines.auto_pipeline import SUPPORTED_TASKS_MAPPINGS, _get_task_class
 37
 38            alt_tasks = set()
 39            for task_map in SUPPORTED_TASKS_MAPPINGS:

 41                if task_class:
 42                    alt_tasks.add(task_class.__name__)
 43                    dbuq(task_class)
 44                for model_code, pipe_class_obj in task_map.items():
 45                    if code_name in model_code:
 46                        alt_tasks.add(pipe_class_obj.__name__)

 51    def show_transformers_tasks(class_name: str | None = None, code_name: str | None = None) -> list[str]:
 52        """Retrieves a list of task classes associated with a specified transformer class.\n
 53        :param class_name: The name of the transformer class to inspect.
 54        :param pkg_type: The dependency for the module
 55        :param alt_method: Use an alternate method to return the classes
 56        :return: A list of task classes associated with the specified transformer."""
 57
 58        task_classes = None
 59
 60        if not code_name:
 61            from mir.config.conversion import import_submodules
 62
 63            class_obj: Callable = import_submodules(class_name, "transformers")
 64            class_module: Callable = import_submodules(*class_obj.__module__.split(".", 1)[-1:], class_obj.__module__.split(".", 1)[0])
 65            if class_module and class_module.__name__ != "DummyPipe":
 66                task_classes = getattr(class_module, "__all__")
 67            else:
 68                return None
 69        elif code_name:
 70            from mir.config.constants import mapped_cls
 71            from httpx import HTTPStatusError
 72            try:
 73                model_class = mapped_cls(code_name)
 74                if model_class is not None:
 75                    # Convert class type to list containing the class name string
 76                    task_classes = [model_class.__name__]
 77                else:
 78                    return None
 79            except (OSError, HTTPStatusError) as e:
 80                dbuq(f"Error mapping class {code_name}: {e}")
 81                return None
 82
 83        return task_classes

 85    async def detect_tasks(self, mir_db: MIRDatabase, field_name: str = "pkg") -> dict:
 86        """Detects and traces tasks MIR data\n
 87        :param mir_db:: An instance of MIRDatabase containing the database of information.
 88        :type mir_db: MIRDatabase
 89        :param field_name:  The name of the field in compatibility data to process for task detection, defaults to "pkg".
 90        :type field_name: str, optional
 91        :return: A dictionary mapping series names to their respective compatibility and traced tasks.
 92        :rtype: dict"""
 93
 94        data_tuple = []
 95        for series, compatibility_data in mir_db.database.items():
 96            if (
 97                series.startswith("info.")  # formatting comment
 98                and not any(tag for tag in self.skip_series if series.startswith(tag))
 99                and not any(tag for tag in self.skip_classes if tag in series)
100            ):
101                for compatibility, field_data in compatibility_data.items():
102                    if field_data and field_data.get(field_name, {}).get("0"):
103                        tasks_for_class = {"tasks": []}
104                        for _, pkg_tree in field_data[field_name].items():
105                            detected_tasks = await self.trace_tasks(pkg_tree=pkg_tree)
106                            if detected_tasks:
107                                for task in detected_tasks:
108                                    if task not in tasks_for_class["tasks"]:
109                                        tasks_for_class["tasks"].append(task)
110                                data_tuple.append((*series.rsplit(".", 1), {compatibility: tasks_for_class}))
111
112        return data_tuple

144    async def hyperlink_to_mir(self, pipe_args: dict, series: str, mir_db: MIRDatabase):
145        """Maps pipeline components to MIR tags/IDs based on class names and roles.\n
146        :param pipe_args: Dictionary of pipeline roles to their corresponding classes
147        :param mir_db: MIRDatabase instance for querying tags/IDs
148        :return: Dictionary mapping pipeline roles to associated MIR tags/IDs"""
149
150        mir_tag: None | list[str] = None
151        detected_links: dict[str, dict] = {"pipe_names": dict()}
152        for pipe_role, pipe_class in pipe_args.items():
153            if pipe_role in ["tokenizer", "tokenizer_2", "tokenizer_3", "tokenizer_4", "prior_tokenizer"]:
154                detected_links["pipe_names"].setdefault(pipe_role, ["info.encoder.tokenizer", series.rsplit(".", 1)[-1]])
155                continue
156            if not any(segment for segment in self.skip_types if pipe_class.__name__ == segment):
157                mir_tag = None
158                detected_links["pipe_names"][pipe_role] = []
159                dbuq(f"pipe_class.__name__ {pipe_class.__name__} {pipe_class}")
160                if pipe_class.__name__ in ["Union"]:
161                    for union_class in pipe_class.__args__:
162                        mir_tag = None
163                        class_name = union_class.__name__
164                        if not any(segment for segment in self.skip_types if class_name == segment):
165                            mir_tag, class_name = await self.tag_class(pipe_class=union_class, pipe_role=pipe_role, series=series, mir_db=mir_db)
166                            # mir_tag = mir_db.find_tag(field="tasks", target=class_name)
167                            # dbuq(f"{mir_tag} {class_name}")
168                        detected_links["pipe_names"][pipe_role].append(mir_tag if mir_tag else class_name)
169                else:
170                    mir_tag, class_name = await self.tag_class(pipe_class=pipe_class, pipe_role=pipe_role, series=series, mir_db=mir_db)
171                    detected_links["pipe_names"][pipe_role] = mir_tag if mir_tag else [class_name]
172                    mir_tag = None
173                    class_name = None
174        return detected_links

176    async def tag_class(self, pipe_class: Callable, pipe_role: str, series: str, mir_db: MIRDatabase) -> tuple[str | None]:
177        """Maps a class to MIR tags/IDs based on its name and role.\n
178        :param pipe_class: Class to be mapped
179        :param pipe_role: Role of the class in the pipeline
180        :param series: Series identifier for the component
181        :param mir_db: MIRDatabase instance for querying tags/IDs
182        :return: Tuple containing MIR tag and class name"""
183
184        from mir.tag import make_scheduler_tag
185
186        mir_tag = None
187        class_name = pipe_class.__name__
188        if pipe_role in ["scheduler", "image_noising_scheduler", "prior_scheduler"]:
189            sub_field = pipe_class.__module__.split(".")[0]
190            scheduler_series, scheduler_comp = make_scheduler_tag(class_name)
191            mir_tag = [f"ops.scheduler.{scheduler_series}", scheduler_comp]
192            if not mir_db.database.get(mir_tag[0], {}).get(mir_tag[1]):
193                mir_tag = mir_db.find_tag(field="pkg", target=class_name, sub_field=sub_field, domain="ops.scheduler")
194            dbuq(f"scheduler {mir_tag} {class_name} {sub_field} ")
195        elif pipe_role == "vae":
196            sub_field = pipe_class.__module__.split(".")[0]
197            mir_comp = series.rsplit(".", 1)[-1]
198            dbuq(mir_comp)
199            mir_tag = [mir_id for mir_id, comp_data in mir_db.database.items() if "info.vae" in mir_id and next(iter(comp_data)) == mir_comp]
200            if mir_tag:
201                mir_tag.append(mir_comp)  # keep mir tag as single list
202            elif class_name != "AutoencoderKL":
203                dbuq(pipe_class)
204                mir_tag = mir_db.find_tag(field="pkg", target=class_name, sub_field=sub_field, domain="info.vae")
205            dbuq(f"vae {mir_tag} {class_name} {sub_field} ")
206        else:
207            mir_tag = mir_db.find_tag(field="tasks", target=class_name)
208        return mir_tag, class_name

210    async def trace_tasks(self, pkg_tree: dict[str, str | int | list[str | int]]) -> List[str]:
211        """Trace tasks for a given MIR entry.\n
212        :param entry: The object containing the model information.
213        :return: A sorted list of tasks applicable to the model."""
214
215        from mir.inspect.classes import resolve_code_names
216
217        preformatted_task_data = None
218        filtered_tasks = None
219        snip_words: set[str] = {"load_tf_weights_in"}
220        package_name = next(iter(pkg_tree))
221        dbuq(pkg_tree)
222        class_name = pkg_tree[package_name]
223        dbuq(f"{package_name}, {class_name}")
224        if class_name not in self.skip_auto:
225            if isinstance(class_name, dict):
226                class_name = next(iter(list(class_name)))
227            if package_name == "transformers":
228                preformatted_task_data = self.show_transformers_tasks(class_name=class_name)
229            elif package_name == "diffusers":
230                code_name = resolve_code_names(class_name, package_name)
231                preformatted_task_data = self.show_diffusers_tasks(code_name=code_name, class_name=class_name)
232                preformatted_task_data.sort()
233            elif package_name == "mflux":
234                preformatted_task_data = self.mflux_tasks
235            if preformatted_task_data:
236                filtered_tasks = [task for task in preformatted_task_data for snip in snip_words if snip not in task]
237                return filtered_tasks  # package_name, class_name

240def trace_classes(pipe_class: str, pkg_name: str) -> Dict[str, List[str]]:
241    """Retrieve all compatible pipe forms\n
242    NOTE: Mainly for Diffusers
243    :param pipe_class: Origin pipe
244    :param pkg_name: Dependency package
245    :return: A dictionary of pipelines"""
246    from mir.inspect.classes import resolve_class_name, extract_inherited
247    from mir.config.conversion import import_submodules
248    from mir.inspect.parenting import class_parent
249
250    related_pipes = []
251    code_name = resolve_class_name(pipe_class, pkg_name)
252    if pkg_name == "diffusers":
253        related_pipe_class_name = pipe_class
254    else:
255        related_pipe_class_name = None
256    related_pipes: list[str] = TaskAnalyzer.show_diffusers_tasks(code_name=code_name, class_name=related_pipe_class_name)
257    # for i in range(len(auto_tasks)):
258    #     auto_tasks.setdefault(i, revealed_tasks[i])
259    parent_folder = class_parent(code_name, pkg_name)
260    if pkg_name == "diffusers":
261        pkg_folder = import_submodules(parent_folder[0], ".".join(parent_folder))
262    else:
263        pkg_folder = import_submodules("__init__", ".".join(parent_folder[:-1]))
264    if hasattr(pkg_folder, "_import_structure"):
265        related_pipes.extend(next(iter(x)) for x in pkg_folder._import_structure.values())
266    related_pipes = set(related_pipes)
267    related_pipes.update(tuple(x) for x in extract_inherited(model_class=pipe_class, pkg_name=pkg_name))
268    return related_pipes
