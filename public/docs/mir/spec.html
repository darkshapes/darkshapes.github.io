122def add_mir_fields(domain: str, **kwargs):
123    """Build MIR dataclasses for each domain type\n
124    :param domain: Class type field constructor
125    :raises ValueError: An invalid domain type was entered
126    :return: A class of the designated type constructed with all fields added
127    """
128    if domain.lower() == "info":
129        return Info(**{k: v for k, v in kwargs.items() if k in Info.__pydantic_fields__ and v is not None})  # pylint: disable=no-member
130    elif domain.lower() == "model":
131        return Model(**{k: v for k, v in kwargs.items() if k in Model.__pydantic_fields__ and v is not None})  # pylint: disable=no-member
132    elif domain.lower() == "ops":
133        return Ops(**{k: v for k, v in kwargs.items() if k in Ops.__pydantic_fields__ and v is not None})  # pylint: disable=no-member
134    elif domain.lower() == "dev":
135        return Dev(**{k: v for k, v in kwargs.items() if k in Dev.__dataclass_fields__ and v is not None})  # pylint: disable=no-member
136    raise ValueError(f"Unsupported domain: {domain}")

139def build_comp(comp: str, domain: str, kwargs: dict) -> Callable:
140    """Create a dynamic Compatibility class and return the defined class object with a flattening function\n
141    :param comp: Name for the class field (and subsequent flattened `key`)
142    :param domain: Class type for the `value` data
143    :param kwargs: Key/value data pairs within `value`
144    :return: A Compatibility object with `comp` name mapped to `kwargs` data
145    """
146    field = {
147        comp: (Union[Info, Model, Ops, Dev], ...),
148    }
149    data = {}
150    data.setdefault(comp, add_mir_fields(domain=domain, **kwargs))
151
152    DynamicModel = create_model("Compatibility", **field)  # pylint: disable=invalid-name
153
154    class Compatibility(DynamicModel):  # pylint: disable=too-few-public-methods
155        """Dynamically created model attributes (to create key structure)"""
156
157        def to_dict(self, _) -> Dict[str, Any]:
158            """Flatten the Compatibility class structure\n
159            :return: A dictionary of the structure
160            """
161            setattr(self, comp, getattr(self, comp).__dict__)  # add the data to an attribute if its not already made
162            return {
163                comp_name: {
164                    inner_key: inner_value
165                    for inner_key, inner_value in comp_value.items()
166                    if inner_value is not None  # Comment to force formatting
167                }
168                for comp_name, comp_value in self.__dict__.items()
169                if isinstance(comp_value, dict)
170            }
171
172    return Compatibility(**data)

175class Series:
176    """
177    Specific seriess of generative and deep learning architectures.\n
178    The release name of models, in short.
179    mir.json contains the lengthy key list of supported seriess\n
180    **add_compat** Add a Compatibility object to this class
181    **to_dict** flatten the compatibility object
182    """
183
184    # @debug_monitor
185    def __init__(self, series: str) -> None:
186        """Constructor"""
187        self.series = series
188        self.compatibility = defaultdict(dict)
189        self.flat_dict = defaultdict(dict)
190
191    def add_compat(self, compat_label: str, compat_obj: Dict[str, int | float | list | str]) -> None:
192        """Add compatibility: Attribute an object to a sub-class of the Series"""
193        self.compatibility[compat_label] = compat_obj
194
195    def to_dict(self, prefix: str) -> Dict[str, Any]:  # , prefix: str = "compatibility")
196        """Flatten the Architecture class structure\n
197        :param prefix: Prepended identifying tag"""
198        for _comp_name, comp_obj in self.compatibility.items():
199            path = f"{prefix}"  # .{comp_name}" # so that we can combine last keys into one dict
200            self.flat_dict[path].update(comp_obj.to_dict(path))  # path
201        return self.flat_dict

206    Known generative and deep learning architecture.\n
207    :param art: Autoregressive transformer, typically LLMs
208    :param dit: Diffusion transformer, typically Vision Synthesis
209    :param lora: Low-Rank Adapter (may work with dit or transformer)
210    :param unet: Unet diffusion structure
211    :param vae: Variational Autoencoder, roughly
212
213    **add_impl** Add an Series object to the Architecture
214    **to_dict** Flatten the Architecture class structure
215    """
216
217    # @debug_monitor
218    def __init__(self, architecture: str) -> None:
219        """Constructor"""
220        self.architecture = architecture
221        self.series = defaultdict(dict)
222        self.flat_dict = defaultdict(dict)
223
224    def add_impl(self, impl_label: str, impl_obj: Series) -> None:
225        """Add_component: Attribute an object to a sub-class of the Architecture"""
226        self.series[impl_label] = impl_obj
227
228    def to_dict(self, prefix: str) -> Dict[str, Any]:
229        """Flatten the Architecture class structure\n
230        :param prefix: Prepended identifying tag"""
231        for comp_name, comp_obj in self.series.items():

237class Domain:
238    """
239    Define a set of AI/ML related data\n
240    :param dev: `Dev()`
241    :param model: `Model()`
242    :param Ops: `Ops()`
243    :param info: `Info()`
244
245    **add_arch** Create a sub-class of Domain\n
246    **to_dict** Flatten the Domain class structure
247    """
248
249    dev: Dev  # Pre-release or under evaluation items without an identifier in an expected format
250    info: Info  # Metadata of layer names or settings with an identifier in the database
251    model: Model  # Model weight specifics of shifting locations and practical dimensions
252    ops: Ops  # References to specific optimization or manipulation techniques
253
254    # @debug_monitor
255    def __init__(self, domain_name: str) -> None:
256        """Constructor"""
257        self.domain_name = domain_name
258        self.architectures = defaultdict(dict)
259        self.flat_dict = defaultdict(dict)
260
261    # @debug_monitor
262    def add_arch(self, arch_label: str, arch_obj: Architecture) -> None:
263        """
264        Add an architecture to subclass this Domain\n
265        :param architecture_name: A valid architecture type
266        :param architecture_obj: Data to store
267        """
268        self.architectures[arch_label] = arch_obj
269
270    # @debug_monitor
271    def to_dict(self) -> Dict[str, Any]:
272        """Flatten the Architecture class structure\n
273        :return: A dictionary of the structure
274        """
275        for arc_name, arc_obj in self.architectures.items():
276            path = f"{self.domain_name}.{arc_name}"
277            self.flat_dict.update(arc_obj.to_dict(path))
278        return self.flat_dict

281def mir_entry(domain: str, arch: str, series: str, comp: str, **kwargs) -> None:
282    """Define a new Machine Intelligence Resource\n
283    :param domain: Broad name of the type of data (model/ops/info/dev)
284    :param arch: Common name of the neural network structure being referenced
285    :param series: Specific release name or technique
286    :param comp: Details about purpose, tasks
287    :param kwargs: Specific key/value data related to location and execution
288    """
289
290    domain_inst = Domain(domain.lower())
291    arch_inst = Architecture(arch.lower())
292    series_inst = Series(series.lower())
293    comp_inst = build_comp(comp, domain, kwargs)
294    series_inst.add_compat(comp, comp_inst)
295    arch_inst.add_impl(series_inst.series, series_inst)
296    domain_inst.add_arch(arch_inst.architecture, arch_inst)
297    return domain_inst.to_dict()
